532:   async importToFossil(
533:     upstreamPath: string,
534:     repoType: "git" | "fossil",
535:     fossilPath: string
536:   ): Promise<VCSResult> {
537:     if (repoType === "git") {
538:       // Import from Git to Fossil using git fast-export
539:       // fossil import --git creates the repo automatically
540:       
541:       // Use git fast-export piped to fossil import
542:       // We need to run: git fast-export --all | fossil import --git fossilPath
543:       const result = await new Promise<{ success: boolean; output: string; error: string }>((resolve) => {
544:         const { spawn } = require("child_process");
545:         
546:         const gitExport = spawn("git", ["fast-export", "--all"], { cwd: upstreamPath });
547:         const fossilImport = spawn("fossil", ["import", "--git", fossilPath], { cwd: upstreamPath });
548:         
549:         let stdout = "";
550:         let stderr = "";
551:         
552:         gitExport.stdout.pipe(fossilImport.stdin);
553:         
554:         fossilImport.stdout.on("data", (data) => {
555:           stdout += data.toString();
556:         });
557:         
558:         fossilImport.stderr.on("data", (data) => {
559:           stderr += data.toString();
560:         });
561:         
562:         gitExport.stderr.on("data", (data) => {
563:           stderr += data.toString();
564:         });
565:         
566:         fossilImport.on("close", (code) => {
567:           resolve({
568:             success: code === 0,
569:             output: stdout,
570:             error: stderr,
571:           });
572:         });
573:         
574:         gitExport.on("error", (err) => {
575:           fossilImport.kill();
576:           resolve({
577:             success: false,
578:             output: "",
579:             error: `Git fast-export failed: ${err.message}`,
580:           });
581:         });
582:       });
583:       
584:       return {
585:         success: result.success,
586:         output: result.output,
587:         error: result.error || undefined,
588:       };
589:     } else {
590:       // For fossil, clone from the upstream fossil repo
591:       const result = await this.execCommand(
592:         ["fossil", "clone", `${upstreamPath}/.fossil`, fossilPath],
593:         upstreamPath
594:       );
595:       
596:       return {
597:         success: result.success,
598:         output: result.output,
599:         error: result.error || undefined,
600:       };
601:     }

811:       const { mkdtempSync } = await import("fs");

813:       const tempDir = mkdtempSync(join(tmpdir(), "mimo-preserve-"));
814:

816:       for (const item of itemsToPreserve) {
817:         const sourcePath = join(upstreamPath, item);
818:         const tempPath = join(tempDir, item);
819:         const isDirectory = statSync(sourcePath).isDirectory();
820:         await this.safeMove(sourcePath, tempPath, isDirectory);

824:       const deleteRecursive = (dirPath: string) => {
825:         if (!fsExistsSync(dirPath)) return;

834:           }
835:         }
836:       };
837:

839: 
840:       // Ensure upstream directory exists
841:       if (!fsExistsSync(upstreamPath)) {
842:         mkdirSync(upstreamPath, { recursive: true });
843:       }
844: 
845:       // Restore preserved items (use safeMove for cross-device support)
846:       for (const item of itemsToPreserve) {
847:         const tempPath = join(tempDir, item);

The issue is that when using `git fast-export --all` with a shallow clone (`--depth 1`), fossil may fail with "short read" errors because the data stream from git is incomplete or doesn't have all the expected data. The error "got 217686 of 9112572 bytes" suggests fossil was expecting more data than git provided.

This typically happens with shallow clones because:
1. `git fast-export --all` expects to export all refs/commits, but with `--depth 1`, many of those references are incomplete
2. Fossil is trying to read a specific amount of data that git can't provide due to the shallow history

The fix would be to either:
1. Do a full clone instead of shallow clone (but that's slow for large repos)
2. Use `--depth` with a larger value that captures enough history
3. Use `git fast-export` with a specific ref instead of `--all` for shallow clones
4. Fall back to creating an empty fossil repo if import fails
